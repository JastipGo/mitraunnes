    if (user) {
        currentUserUid = user.uid;
        await loadDriverProfile();

window.doLogin = async () => {
    const e = document.getElementById('login-email').value, p = document.getElementById('login-pass').value;

async function loadDriverProfile() {
    try {
        const snap = await getDoc(doc(db, "user", currentUserUid));
        if (snap.exists() && snap.data().role === 'driver') {
            driverData = snap.data();
            document.getElementById('prof-nama').innerText = driverData.nama;
            document.getElementById('prof-rank').innerText = driverData.status.toUpperCase();
            
            const saldoSnap = await getDoc(doc(db, "saldo", currentUserUid));
            if(saldoSnap.exists()) {
                document.getElementById('top-saldo').innerText = "Rp " + saldoSnap.data().saldo.toLocaleString('id-ID');
            }

            document.getElementById('splash').classList.add('hidden');
            document.getElementById('app').classList.add('visible');
            
            listenToOrders();
            hideLoading();
        } else {
            alert("Akun ini bukan Driver!");
            signOut(auth);
            hideLoading();
        }
    } catch(e) { console.error(e); hideLoading(); }
}

// LISTENER ORDER (LOGIKA SAMA, UI BARU)

function listenToOrders() {
    // 1. Order Tersedia
    const qAvailable = query(collection(db, "orders"), where("getdriverId", "==", ""), where("status", "==", "menunggu"));
    onSnapshot(qAvailable, (snapshot) => {
        let html = '';
        snapshot.forEach(doc => {
            const o = doc.data();
            html += `
            <div class="card">
                <div class="card-header">
                    <div>
                        <div class="price">Rp ${o.Argo.toLocaleString('id-ID')}</div>
                        <div style="font-size:13px; color:var(--gray-300); margin-top:4px;">${o.kategori.toUpperCase()} • ${o.JarakTempuh} KM</div>
                    </div>
                </div>
                <div class="location-row">
                    <span class="material-symbols-outlined loc-icon">radio_button_checked</span>
                    <div>
                        <div class="loc-text">${o.namaAmbil}</div>
                        <div class="loc-sub">Titik Pengambilan</div>
                    </div>
                </div>
                <div class="location-row" style="margin-bottom:20px;">
                    <span class="material-symbols-outlined loc-icon" style="color:var(--black);">location_on</span>
                    <div>
                        <div class="loc-text">${o.NamaAntar}</div>
                        <div class="loc-sub">Tujuan Pengantaran</div>
                    </div>
                </div>
                
                <div class="swipe-container" data-action="terima" data-orderid="${doc.id}">
                    <div class="swipe-text">Geser untuk Terima</div>
                    <div class="swipe-knob"><span class="material-symbols-outlined">arrow_forward</span></div>
                </div>
            </div>`;
        });
        document.getElementById('available-orders').innerHTML = html || '<div style="padding:40px 20px;text-align:center;color:var(--gray-300);"><span class="material-symbols-outlined" style="font-size:48px;margin-bottom:16px;">search</span><p>Mencari permintaan...</p></div>';
        initSwipeInteractions();
    });

    // 2. Order Aktif
    const qActive = query(collection(db, "orders"), where("getdriverId", "==", currentUserUid));
    onSnapshot(qActive, (snapshot) => {
        let activeHTML = '';
        let hasActive = false;
        activeOrderDocId = null;

        snapshot.forEach(d => {
            const o = d.data();
            if (o.status !== 'selesai' && o.status !== 'dibatalkan') {
                hasActive = true;
                activeOrderDocId = d.id;
                activeHTML = renderActiveOrder(d.id, o);
                
                if(o.status === 'ngantar' && !locationInterval) {
                    startLocationTracking(d.id);
                } else if (o.status !== 'ngantar' && locationInterval) {
                    clearInterval(locationInterval);
                    locationInterval = null;
                }
            }
        });

        const navMaps = document.getElementById('nav-maps');
        if(hasActive) {
            document.getElementById('active-order-container').innerHTML = activeHTML;
            document.getElementById('available-orders').innerHTML = '<div style="padding:20px; text-align:center; color:var(--gray-300);">Anda sedang dalam pesanan berjalan.</div>';
            
            // Buka akses Maps dan pindah kesana
            navMaps.classList.remove('disabled');
            if(!document.getElementById('page-maps').classList.contains('active')) {
                window.switchPage('maps');
            }
            initSwipeInteractions();
        } else {
            document.getElementById('active-order-container').innerHTML = '';
            // Tutup akses maps
            navMaps.classList.add('disabled');
            if(document.getElementById('page-maps').classList.contains('active')) {
                window.switchPage('cari');
            }
            if(locationInterval) { clearInterval(locationInterval); locationInterval = null; }
        }
    });
}

    const swipers = document.querySelectorAll('.swipe-container');
    swipers.forEach(container => {
        if(container.dataset.initialized) return; // hindari duplicate event
        container.dataset.initialized = "true";

        const knob = container.querySelector('.swipe-knob');
        let isDragging = false, startX = 0, currentX = 0;
        const maxDrag = container.offsetWidth - knob.offsetWidth - 8; // 8px padding (4px kiri, 4px kanan margin)

        const onMove = (e) => {
            if(!isDragging) return;
            const x = (e.touches ? e.touches[0].clientX : e.clientX) - startX;
            currentX = Math.max(0, Math.min(x, maxDrag));
            knob.style.transform = `translateX(${currentX}px)`;
        };

        const onEnd = () => {
            if(!isDragging) return;
            isDragging = false;
            knob.style.transition = 'transform 0.3s ease';
            
            if(currentX >= maxDrag * 0.85) { // Threshold 85% diswipe
                knob.style.transform = `translateX(${maxDrag}px)`;
                executeSwipeAction(container);
            } else {
                knob.style.transform = `translateX(0px)`;
            }
        };

        const onStart = (e) => {
            isDragging = true;
            startX = (e.touches ? e.touches[0].clientX : e.clientX) - currentX;
            knob.style.transition = 'none';
        };

        knob.addEventListener('touchstart', onStart, {passive: true});
        document.addEventListener('touchmove', onMove, {passive: true});
        document.addEventListener('touchend', onEnd);
    });
}

function executeSwipeAction(el) {
    const action = el.dataset.action;
    const orderId = el.dataset.orderid;

    if (action === 'terima') {
        window.terimaOrder(orderId);
    } else if (action === 'status') {
        window.updateStatus(orderId, el.dataset.status);
    } else if (action === 'uploadAmbil') {
        window.uploadBuktiAmbil(orderId);
    } else if (action === 'uploadSelesai') {
        window.uploadBuktiSelesai(orderId, parseInt(el.dataset.argo));
    }
}

async function uploadImageToCloudinary(file) {
    const formData = new FormData(); formData.append('file', file); formData.append('upload_preset', CLOUDINARY_UPLOAD_PRESET);
    try {
        const res = await fetch(`https://api.cloudinary.com/v1_1/${CLOUDINARY_CLOUD_NAME}/image/upload`, { method: 'POST', body: formData });
        const data = await res.json();
        return data.secure_url; 
    } catch (e) {
        return new Promise((resolve) => {
            const reader = new FileReader(); reader.onload = (ev) => resolve(ev.target.result); reader.readAsDataURL(file);
        });
    }
}

function getCoordinates() {
    return new Promise((resolve, reject) => {
        if(!navigator.geolocation) return reject("GPS tidak didukung");
        navigator.geolocation.getCurrentPosition(
            p => resolve(`${p.coords.longitude.toFixed(6)}, ${p.coords.latitude.toFixed(6)}`),
            e => reject(e.message), { enableHighAccuracy: true }
        );
    });
}

window.uploadBuktiAmbil = async (orderId) => {
    if(!fileBuffer) { alert('Pilih foto barang terlebih dahulu sebelum swipe!'); return resetSwipe(); }
    showLoading('Mengupload Bukti & GPS...');
    try {
        let koordinat = "";
        try { koordinat = await getCoordinates(); } catch(g) { console.log('Gagal GPS', g); }
        const imgUrl = await uploadImageToCloudinary(fileBuffer); fileBuffer = null;
        await updateDoc(doc(db, "orders", orderId), { titikAmbil: koordinat, urlbuktiAmbil: imgUrl, status: "ngantar" });
    } catch(e) { alert('Gagal: ' + e.message); resetSwipe(); } finally { hideLoading(); }
};

window.uploadBuktiSelesai = async (orderId, argoAmount) => {
    if(!fileBuffer) { alert('Pilih foto bukti selesai terlebih dahulu sebelum swipe!'); return resetSwipe(); }
    showLoading('Menyelesaikan Pesanan...');
    try {
        const imgUrl = await uploadImageToCloudinary(fileBuffer); fileBuffer = null;
        await updateDoc(doc(db, "orders", orderId), { urlbuktiSelesai: imgUrl, status: "selesai" });
        await setDoc(doc(collection(db, "pendapatan")), { createdBy: currentUserUid, uang: argoAmount, timestamp: new Date().toISOString() });
        const saldoRef = doc(db, "saldo", currentUserUid);
        const saldoSnap = await getDoc(saldoRef);
        let newSaldo = argoAmount; if(saldoSnap.exists()) newSaldo += saldoSnap.data().saldo;
        await updateDoc(saldoRef, { saldo: newSaldo });
        loadDriverProfile(); 
    } catch(e) { alert('Gagal: ' + e.message); resetSwipe(); } finally { hideLoading(); }
};

function resetSwipe() {
    // Utility untuk mereset knob swipe jika upload gagal/belum pilih file
    document.querySelectorAll('.swipe-knob').forEach(k => { k.style.transform = 'translateX(0px)'; });
}

function startLocationTracking(orderId) {
    if(locationInterval) clearInterval(locationInterval);
    const pushLocation = () => {
        if(!navigator.geolocation) return;
        navigator.geolocation.getCurrentPosition(async (p) => {
            let coords = `${p.coords.longitude.toFixed(6)}, ${p.coords.latitude.toFixed(6)}`;
            try { await updateDoc(doc(db, "orders", orderId), { sharelocationDriver: coords }); } catch(e) {}
        }, null, { enableHighAccuracy: true });
    };
    pushLocation(); 
    locationInterval = setInterval(pushLocation, 30000); 
}
