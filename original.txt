// GPS TRACKER
function trackDeviceLocation() {
    if(!navigator.geolocation) return;
    navigator.geolocation.watchPosition(p => {
        currentDeviceCoords = [p.coords.longitude, p.coords.latitude];
        if(map && map.getSource('driver-loc')) {
            map.getSource('driver-loc').setData({ type: 'Point', coordinates: currentDeviceCoords });
        }
    }, null, { enableHighAccuracy: true });
}

function parseCoords(str) {
    if(!str) return null;
    let m = str.match(/(-?\d+\.\d+),\s*(-?\d+\.\d+)/);
    if(!m) return null;
    let v1 = parseFloat(m[1]), v2 = parseFloat(m[2]);
    let lng = (Math.abs(v1) > 90) ? v1 : v2;
    let lat = (Math.abs(v1) > 90) ? v2 : v1;
    return [lng, lat];
}

// HAVERSINE DISTANCE
function getDistance(lat1, lon1, lat2, lon2) {
    const R = 6371; const dLat = (lat2 - lat1) * Math.PI / 180; const dLon = (lon2 - lon1) * Math.PI / 180;
    const a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2);
    return R * (2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)));
}

function listenToOrders() {
    // 1. ACTIVE ORDERS (Driver's Orders)
    const qActive = query(collection(db, "orders"), where("getdriverId", "==", currentUserUid));
    onSnapshot(qActive, (snapshot) => {
        let tempActive = [];
        snapshot.forEach(d => {
            const o = d.data();
            if(o.status !== 'selesai' && o.status !== 'dibatalkan') tempActive.push({ id: d.id, ...o });
        });
        activeOrders = tempActive;
        renderActiveOrdersPhase();
    });

    // 2. AVAILABLE ORDERS
    const qAvailable = query(collection(db, "orders"), where("getdriverId", "==", ""), where("status", "==", "menunggu"));
    onSnapshot(qAvailable, (snapshot) => {
        let html = '';
        const now = new Date().getTime();
        
        // Pengecekan Batching (Jika sedang bawa Normal, atau Gabungan sudah 4)
        let isCarryingNormal = activeOrders.some(o => o.jenisOrder === 'normal');
        let gabunganCount = activeOrders.filter(o => o.jenisOrder === 'gabungan').length;
        if(isCarryingNormal || gabunganCount >= 4) {
            document.getElementById('available-orders').innerHTML = '<div style="padding:40px 20px;text-align:center;color:var(--gray-300);">Anda sudah mencapai batas pesanan aktif.</div>';
            return;
        }

        snapshot.forEach(doc => {
            const o = doc.data();
            // FILTER WAKTU: Tampilkan yang <= 60 menit dari jadwal, atau Langsung.
            let showOrder = false;
            if (o.waktuAmbil === 'Langsung') {
                showOrder = true; 
            } else {
                let sched = new Date(o.waktuAmbil).getTime();
                let diffMins = (sched - now) / (1000 * 60);
                if (diffMins <= 60 && diffMins >= -120) showOrder = true; // Batas toleransi jadwal
            }

            // FILTER BATCHING GABUNGAN: Hanya tampilkan gabungan jika driver sedang bawa gabungan.
            if(gabunganCount > 0 && o.jenisOrder !== 'gabungan') showOrder = false;

            if (showOrder) {
                html += `
                <div class="card">
                    <div class="card-header">
                        <div>
                            <div class="price">Rp ${o.Argo.toLocaleString('id-ID')}</div>
                            <div style="font-size:13px; color:var(--gray-300); margin-top:4px;">
                                <span class="badge-order" style="background:${o.jenisOrder==='gabungan'?'#E6F6ED':'var(--gray-200)'}; color:${o.jenisOrder==='gabungan'?'var(--green)':'var(--black)'};">${o.jenisOrder}</span> • ${o.kategori} • ${o.JarakTempuh} KM
                            </div>
                        </div>
                    </div>
                    <div class="location-row"><span class="material-symbols-outlined loc-icon">radio_button_checked</span><div><div class="loc-text">${o.namaAmbil}</div><div class="loc-sub">Ambil</div></div></div>
                    <div class="location-row" style="margin-bottom:20px;"><span class="material-symbols-outlined loc-icon" style="color:var(--black);">location_on</span><div><div class="loc-text">${o.NamaAntar}</div><div class="loc-sub">Antar</div></div></div>
                    <div class="swipe-container" data-action="terima" data-orderid="${doc.id}">
                        <div class="swipe-text">Geser untuk Terima</div>
                        <div class="swipe-knob"><span class="material-symbols-outlined">arrow_forward</span></div>
                    </div>
                </div>`;
            }
        });
        if(!html) html = '<div style="padding:40px 20px;text-align:center;color:var(--gray-300);"><span class="material-symbols-outlined" style="font-size:48px;margin-bottom:16px;">search</span><p>Mencari permintaan...</p></div>';
        document.getElementById('available-orders').innerHTML = html;
        initSwipeInteractions();
    });
}

            </div>`;
        } else {
            // UPLOAD FOTO UNTUK SETIAP PESANAN, LALU 1 SWIPE NGANTAR
            group.forEach((o, i) => {
                html += `
                <div style="background:var(--gray-100); padding:12px; border-radius:12px; margin-bottom:12px;">
                    <div style="font-weight:600; margin-bottom:4px;">${i+1}. Penerima: ${o.namaPenerima}</div>
                    <div style="font-size:12px; color:var(--gray-800); margin-bottom:8px;">Catatan: ${o.catatan || '-'}</div>
                    <div class="upload-box" style="background:#FFF;">
                        <input type="file" accept="image/*" onchange="window.prepareBatchFile(event, '${o.id}', 'ambil')">
                        <div class="upload-box-content">
                            <span class="material-symbols-outlined" style="font-size:20px;">photo_camera</span>
                            <span id="label-${o.id}-ambil">Ketuk Foto Barang/Nota</span>
                        </div>
                    </div>
                </div>`;
            });
            html += `
            <div class="swipe-container" data-action="batch_ngantar" data-orderids="${orderIds}">
                <div class="swipe-text">Geser Mulai Antar Semua</div>
                <div class="swipe-knob"><span class="material-symbols-outlined">two_wheeler</span></div>
            </div>`;
        }
        container.innerHTML = html;
        initSwipeInteractions();
        return;
    }

    // PHASE 2: DELIVERY (Semua order sudah ngantar/tiba)
    enableMap();
    let deliveries = activeOrders.filter(o => o.status === 'ngantar' || o.status === 'tiba');
    
    // SORT BERDASARKAN JARAK TERDEKAT DARI LOKASI SAAT INI (Jika ada GPS)
    let refCoords = currentDeviceCoords;
    if(!refCoords) refCoords = parseCoords(deliveries[0].titikAmbil); // fallback

    if(refCoords) {
        deliveries.sort((a, b) => {
            let cA = parseCoords(a.mpasAntar); let cB = parseCoords(b.mpasAntar);
            if(!cA) return 1; if(!cB) return -1;
            return getDistance(refCoords[1], refCoords[0], cA[1], cA[0]) - getDistance(refCoords[1], refCoords[0], cB[1], cB[0]);
        });
    }

    // Ambil PIN 1 (Yang paling dekat & belum selesai)
    currentDeliveryPin = deliveries[0];
    drawRouteOSRM(); // Gambar rute ke Pin 1

    let o = currentDeliveryPin;
    let html = `
    <div style="display:flex; justify-content:space-between; align-items:center; margin-bottom:12px;">
        <h3 style="font-size:20px;">Pengantaran Saat Ini (Pin 1)</h3>
        <div style="background:var(--gray-200); padding:4px 8px; border-radius:4px; font-size:12px; font-weight:700;">${o.jenisOrder.toUpperCase()}</div>
    </div>
    <div class="location-row" style="margin-bottom:16px;">
        <span class="material-symbols-outlined loc-icon">location_on</span>
        <div><div class="loc-text">${o.NamaAntar}</div><div class="loc-sub">Penerima: ${o.namaPenerima} (${o.waPenerima})</div></div>
    </div>`;

    if(o.status === 'ngantar') {
        html += `
        <div class="swipe-container action-green" data-action="tiba_tujuan" data-orderid="${o.id}">
            <div class="swipe-text" style="color:var(--green);">Geser Tiba di Tujuan</div>
            <div class="swipe-knob"><span class="material-symbols-outlined">flag</span></div>
        </div>`;
    } else if (o.status === 'tiba') {
        html += `
        <div class="upload-box">
            <input type="file" accept="image/*" onchange="window.prepareBatchFile(event, '${o.id}', 'selesai')">
            <div class="upload-box-content">
                <span class="material-symbols-outlined">verified</span>
                <span id="label-${o.id}-selesai">Ketuk Foto Bukti Selesai</span>
            </div>
        </div>
        <div class="swipe-container" style="background:#000;" data-action="selesai_pesanan" data-orderid="${o.id}" data-argo="${o.Argo}">
            <div class="swipe-text" style="color:#FFF;">Geser Selesai & Terima Rp${o.Argo.toLocaleString()}</div>
            <div class="swipe-knob" style="background:#FFF;color:#000;"><span class="material-symbols-outlined">check</span></div>
        </div>`;
    }
    
    // Tampilkan order selanjutnya sebagai antrean
    if(deliveries.length > 1) {
        html += `<hr style="border:none; border-top:1px solid var(--gray-200); margin:16px 0;"><h4 style="font-size:14px; color:var(--gray-300); margin-bottom:8px;">Antrean Selanjutnya:</h4>`;
        for(let i=1; i<deliveries.length; i++) {
            html += `<div style="font-size:13px; color:var(--gray-800); margin-bottom:4px;">Pin ${i+1}: ${deliveries[i].namaPenerima} - ${deliveries[i].NamaAntar}</div>`;
        }
    }
    container.innerHTML = html;
    initSwipeInteractions();
}


// MAPS LOGIC (OSRM)
function disableMap() {
    document.getElementById('map').classList.add('d-none');
    document.getElementById('map-placeholder').style.display = 'flex';
}
function enableMap() {
    if(!activeOrders.some(o => o.kategori !== 'Kirim Instan')) return; // Map hanya untuk jastip/ambil instan
    document.getElementById('map').classList.remove('d-none');
    document.getElementById('map-placeholder').style.display = 'none';
    if(!map) {
        map = new maplibregl.Map({ container: 'map', style: 'https://tiles.openfreemap.org/styles/bright', center: currentDeviceCoords || [110.3939, -7.0515], zoom: 15, attributionControl: false });
        map.on('load', () => {
            // Add Motor Icon Source
            map.addSource('driver-loc', { type: 'geojson', data: { type: 'Point', coordinates: currentDeviceCoords || [0,0] } });
            map.addLayer({ id: 'driver-layer', type: 'circle', source: 'driver-loc', paint: { 'circle-radius': 8, 'circle-color': '#000', 'circle-stroke-width': 2, 'circle-stroke-color': '#FFF' } });
        });
    }
}
async function drawRouteOSRM() {
    if(!map || !currentDeliveryPin) return;
    let start = currentDeviceCoords || parseCoords(currentDeliveryPin.titikAmbil);
    let end = parseCoords(currentDeliveryPin.mpasAntar);
    if(!start || !end) return;

    try {
        const url = `https://router.project-osrm.org/route/v1/driving/${start[0]},${start[1]};${end[0]},${end[1]}?overview=full&geometries=geojson`;
        const res = await fetch(url);
        const data = await res.json();
        if(data.routes && data.routes[0]) {
            const geojson = data.routes[0].geometry;
            if(map.getSource('route')) map.getSource('route').setData(geojson);
            else {
                map.addSource('route', { type: 'geojson', data: geojson });
                map.addLayer({ id: 'route-layer', type: 'line', source: 'route', layout: { 'line-join': 'round', 'line-cap': 'round' }, paint: { 'line-color': '#05A357', 'line-width': 5 } });
            }
            
            // Add End Pin
            if(map.getSource('pin-end')) map.getSource('pin-end').setData({ type: 'Point', coordinates: end });
            else {
                map.addSource('pin-end', { type: 'geojson', data: { type: 'Point', coordinates: end } });
                map.addLayer({ id: 'pin-end-layer', type: 'circle', source: 'pin-end', paint: { 'circle-radius': 10, 'circle-color': '#E11900', 'circle-stroke-width': 3, 'circle-stroke-color': '#FFF' } });
            }
            map.fitBounds([start, end], { padding: 50 });
        }
    } catch(e) { console.error("OSRM Error", e); }
}


// SWIPE GESTURES
window.prepareBatchFile = (e, orderId, type) => {
    if(e.target.files.length > 0) {
        window.filesBuffer[orderId + '_' + type] = e.target.files[0];
        document.getElementById(`label-${orderId}-${type}`).innerText = "File Siap";
        document.getElementById(`label-${orderId}-${type}`).style.color = "var(--green)";
    }
};

function initSwipeInteractions() {
    document.querySelectorAll('.swipe-container').forEach(container => {
        if(container.dataset.initialized) return;
        container.dataset.initialized = "true";
        const knob = container.querySelector('.swipe-knob');
        let isDragging = false, startX = 0, currentX = 0;
        const maxDrag = container.offsetWidth - knob.offsetWidth - 8;

        const onMove = (e) => {
            if(!isDragging) return;
            const x = (e.touches ? e.touches[0].clientX : e.clientX) - startX;
            currentX = Math.max(0, Math.min(x, maxDrag));
            knob.style.transform = `translateX(${currentX}px)`;
        };
        const onEnd = () => {
            if(!isDragging) return;
            isDragging = false;
            knob.style.transition = 'transform 0.3s ease';
            if(currentX >= maxDrag * 0.85) {
                knob.style.transform = `translateX(${maxDrag}px)`;
                executeSwipeAction(container);
            } else { knob.style.transform = `translateX(0px)`; }
        };
        const onStart = (e) => {
            isDragging = true; startX = (e.touches ? e.touches[0].clientX : e.clientX) - currentX; knob.style.transition = 'none';
        };
        knob.addEventListener('touchstart', onStart, {passive: true});
        document.addEventListener('touchmove', onMove, {passive: true});
        document.addEventListener('touchend', onEnd);
    });
}
function resetSwipe() { document.querySelectorAll('.swipe-knob').forEach(k => { k.style.transform = 'translateX(0px)'; }); }

async function uploadImage(file) {
    const fd = new FormData(); fd.append('file', file); fd.append('upload_preset', CLOUDINARY_UPLOAD_PRESET);
    try {
        const res = await fetch(`https://api.cloudinary.com/v1_1/${CLOUDINARY_CLOUD_NAME}/image/upload`, { method: 'POST', body: fd });
        const data = await res.json(); return data.secure_url; 
    } catch (e) {
        return new Promise(r => { const rd = new FileReader(); rd.onload = ev => r(ev.target.result); rd.readAsDataURL(file); });
    }
}

// BACKEND ACTIONS
async function executeSwipeAction(el) {
    const action = el.dataset.action;
    const orderIds = el.dataset.orderids ? el.dataset.orderids.split(',') : [];
    const orderId = el.dataset.orderid;

    try {
        showLoading('Memproses...');
        if (action === 'terima') {
            await updateDoc(doc(db, "orders", orderId), { getdriverId: currentUserUid, driverName: driverData.nama, waDriver: driverData.wa, sinceDriver: "Mitra Aktif", rankDriver: driverData.status, UrlprofileDriver: driverData.urlProfile || "", kendaraanDriver: driverData.stnk || "-", status: "terima" });
        } 
        else if (action === 'batch_ambil') {
            await Promise.all(orderIds.map(id => updateDoc(doc(db, "orders", id), { status: "ambil" })));
        } 
        else if (action === 'batch_ngantar') {
            // Validasi file
            for(let id of orderIds) {
                if(!window.filesBuffer[id + '_ambil']) { alert('Lengkapi semua foto pesanan!'); hideLoading(); resetSwipe(); return; }
            }
            let koordinat = currentDeviceCoords ? `${currentDeviceCoords[0]},${currentDeviceCoords[1]}` : "";
            // Upload & Update concurrent
            await Promise.all(orderIds.map(async (id) => {
                let imgUrl = await uploadImage(window.filesBuffer[id + '_ambil']);
                return updateDoc(doc(db, "orders", id), { titikAmbil: koordinat, urlbuktiAmbil: imgUrl, status: "ngantar" });
            }));
            // Bersihkan buffer
            orderIds.forEach(id => delete window.filesBuffer[id + '_ambil']);
        }
        else if (action === 'tiba_tujuan') {
            await updateDoc(doc(db, "orders", orderId), { status: "tiba" });
        }
        else if (action === 'selesai_pesanan') {
            if(!window.filesBuffer[orderId + '_selesai']) { alert('Pilih foto bukti selesai!'); hideLoading(); resetSwipe(); return; }
            let imgUrl = await uploadImage(window.filesBuffer[orderId + '_selesai']);
            await updateDoc(doc(db, "orders", orderId), { urlbuktiSelesai: imgUrl, status: "selesai" });
            
            // Tambah Pendapatan Cache
            addIncome(parseInt(el.dataset.argo));
            delete window.filesBuffer[orderId + '_selesai'];
        }
        hideLoading();
    } catch(e) {
        console.error(e);
        alert('Terjadi kesalahan: ' + e.message);
        hideLoading();
        resetSwipe();
    }
}
